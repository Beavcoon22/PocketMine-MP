public function ban($username)
	Description:
		Bans a username
	Parameters:
		$username = username to ban
	Return Values:
		None

public function pardon($username)
	Description:
		Unbans a username
	Parameters:
		$username = username to ban
	Return Values:
		None
		
public function banIP($ip)
	Description:
		IPbans an IP
	Parameters:
		$ip = ip to ban
	Return Values:
		None
		
public function pardonIP($ip)
	Description:
		IPpardons an IP
	Parameters:
		$ip = ip to pardon
	Return Values:
		None
		
public function kick($username, $reason)
	Description:
		Kicks a person out of the server
	Parameters:
		$username = username of a player to kick
		$reason = reason for kicking the player
	Return Values:
		None

public function reload()
	Description:
		Reloads the ban configuration files
	Parameters:
		None
	Return Values:
		None
